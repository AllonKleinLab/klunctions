This notebook uses custom functions to:

  1. Load raw counts matrices (from indrops.py pipeline) for multiple libraries
  2. Filter out background barcodes (i.e., free-floating RNA from otherwise empty droplets)
  3. Filter out dead cells (high in mitochondrial genes)
  4. Merge data from multiple libraries
  5. Perform standard pre-processing:
    a. Normalization
    b. Highly variable gene filtering
    c. Centering and variance scaling (z-score)
    d. PCA
    e. KNN graph construction
    f. Louvain clustering g. UMAP (2-D embedding)
    h. Doublet prediction
  6. Export to SPRING plot for interactive data exploration

Load counts matrices¶

In [4]:
# D is a dictionary that will store data for each library (sample). 
# Each entry (one per library) is another dictionary with the 
# following keys (loosely based on AnnData scheme):
#   - E: raw counts matrix (scipy.sparse.csc_matrix, shape (n_cells, n_genes))
#   - obs: metadata for each cell (pandas dataframe, shape (n_cells, n_annotations))
#   - uns: for unstructured data/parameters (dictionary)

D = {}

#####
# Set some default filtering parameters. These can be
# customized for each library at a later step.

# min number of UMIs per cell barcode, for separating 
# background from real cells
default_min_tot = 500  

# max fraction of UMIs from mitochondrial genes, for 
# separating live and dead cells
default_max_mito_frac = 0.15 

####
# Loop over each library, adding data to D

for s in sample_name:
    # Initialize entry with default filtering parameters
    D[s] = {'uns': {'min_tot': default_min_tot, 'max_mt': default_max_mito_frac}}
    
    # Try loading counts matrix from .npz file (fast).
    # If .npz file doesn't exist yet, load from .tsv file
    # (the output of indrops.py pipeline) and save .npz 
    # for future use.
    npz_file = f'{data_dir}/{s}.counts.npz'
    
    if os.path.exists(npz_file):
        # Load counts from .npz file, as well as cell barcode IDs and gene names
        Etmp = ssp.load_npz(npz_file)
        bctmp = np.loadtxt(f'{data_dir}/{s}.barcodes.txt', dtype=str)
        genetmp = np.loadtxt(f'{data_dir}/{s}.genes.txt', dtype=str)
    
    else:
        # Load counts matrix, cell barcode IDs, and gene names from indrops.py .tsv file
        Etmp, bctmp, genetmp = hf.load_annotated_text(hf.file_opener(f'{data_dir}/{s}.counts.tsv.gz'), 
                                                      read_column_labels=True, read_row_labels=True)

        # Save counts matrix, cell barcode IDs, and gene names to files for future use.
        ssp.save_npz(npz_file, Etmp, compressed=True)
        np.savetxt(f'{data_dir}/{s}.barcodes.txt', bctmp, fmt='%s')
        np.savetxt(f'{data_dir}/{s}.genes.txt', genetmp, fmt='%s')
    
    # Store data in data dictionary
    D[s]['E'] = Etmp.copy()
    
    # Initialize obs dataframe with barcode IDs, library ID,
    # and any other pre-specified metadata from meta_labels
    D[s]['obs'] = pd.DataFrame({'barcode': bctmp})
    D[s]['obs']['library'] = s
    for iLab, lab in enumerate(meta_labels):
        D[s]['obs'][lab] = meta_info[s][iLab]
        
    # All libraries have same gene list, so just store as global variable
    gene_list = genetmp.copy()
    
    # Print info about what we loaded
    print('{:20} {} cell barcodes, {} genes'.format(s, *D[s]['E'].shape))
    
    # Clean up temporary variables
    del Etmp, bctmp, genetmp

Filter out background based on UMIs/cell¶

First, plot histograms and customize filtering thresholds¶

No filtering is actually done here, just checking that the total UMIs threshold is appropriate for each library. The threshold should lie between the two modes of the histogram, where the left mode corresponds to background and the right to real cells. Also check that (roughly) the expected number of cells are passing the filter.

In [5]:
# Change total count thresholds based on the histograms. 
# If a library is not included below, the default threshold is used.
D['d10_ctrl_1']['uns']['min_tot'] = 750
D['d30_ctrl_2']['uns']['min_tot'] = 300
D['m4_ctrl_1']['uns']['min_tot'] = 300
D['m4_ctrl_2']['uns']['min_tot'] = 300

# Start figure
fig, nrow, ncol = hf.start_subplot_figure(len(sample_name), row_height=2.5)

for iS,s in enumerate(sample_name):
    # Loop over each library, plotting histogram of total counts (UMIs) per cell
    
    # Set subplot
    ax = plt.subplot(nrow, ncol, iS+1)
    
    # Calculate total counts per cells (row sums of counts matrix)
    D[s]['obs']['total_counts'] = D[s]['E'].sum(1).A.squeeze()
    
    # Get minimum total counts threshold. Barcodes with 
    # total counts >= min_tot are considered real cells. 
    min_tot = D[s]['uns']['min_tot']
    
    # Plot the histogram
    hf.total_counts_histogram(D[s]['obs']['total_counts'], min_bin=10, max_bin=3e4, ax=ax)
    ax.set_title(s, fontsize=12)
    
    xl = np.array(ax.get_xlim())
    yl = np.array(ax.get_ylim())
    ax.set_xlim(xl)
    ax.set_ylim(yl)
    
    # Plot the threshold    
    ax.plot(min_tot * np.ones(2), yl, c='black', linewidth=1)
    
    # Calculate number of barcodes passing filter and show on plot
    ntot = len(D[s]['obs']['total_counts'])
    npass = (D[s]['obs']['total_counts'] >= min_tot).sum()
    ax.text(xl[0] + 10**(np.log10(xl.ptp())*0.05), yl[0] + yl.ptp() * 0.88, '{} / {}'.format(npass, ntot), fontsize=12)
    ax.text(xl[0] + 10**(np.log10(xl.ptp())*0.05), yl[0] + yl.ptp() * 0.77, '{:.1f}%'.format(npass/ntot*100), fontsize=12)
    
fig.tight_layout()

min_vscore_pctl = 85

# Number of principal components
n_pcs = 50

# Number of neighbors for KNN graph construction
n_neighbors = 5

# UMAP parameters
umap_n_neighbors = 10
umap_min_dist = 0.3

Run pre-processing¶

In [12]:
# Total counts normalization to help remove 
# technical variation between cells
print('Normalizing...')
Dm['E_norm'] = hf.tot_counts_norm(Dm['E'])[0]

# Identify highly variable genes
print('Finding highly variable genes...')
Dm['uns']['gene_filter'] = hf.filter_genes(Dm['E_norm'], 
                                           min_counts=min_counts, 
                                           min_cells=min_cells, 
                                           min_vscore_pctl=min_vscore_pctl, 
                                           show_vscore_plot=True)

# Run PCA on cells using highly variable genes. 
# hf.get_pca automatically scales the data (z-score) 
# at the gene level before running PCA.
print('Using {} genes for PCA...'.format(len(Dm['uns']['gene_filter'])))
Dm['pca'] = hf.get_pca(Dm['E_norm'][:, Dm['uns']['gene_filter']], 
                       numpc=n_pcs)

# Build k-nearest neighbor graph
print('Getting kNN graph...')
Dm['uns']['knn_edges'] = hf.get_knn_graph(Dm['pca'], k=n_neighbors, return_edges=True)[0]

# Cluster cells (Louvain clustering)
print('Running Louvain clustering...')
Dm['obs']['louvain'] = hf.get_louvain_clusters(Dm['E'].shape[0], Dm['uns']['knn_edges'], resolution=1.0)

# Get 2-D embedding using UMAP
print('Running UMAP...')
Dm['umap'] = hf.get_umap(Dm['pca'], 
                         min_dist=umap_min_dist, 
                         n_neighbors=umap_n_neighbors
                        )

print('Done!')
